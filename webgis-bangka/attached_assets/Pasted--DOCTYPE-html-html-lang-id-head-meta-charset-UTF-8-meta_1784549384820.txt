    <script>
        //==========================================================
        // 1. INITIALISASI PETA DI KABUPATEN BANGKA BARAT
        //==========================================================
        var map = L.map('map', {
            zoomControl: false
        }).setView([-1.83, 105.45], 10); // Sentralisasi koordinat Bangka Barat

        // Tambahkan tombol Zoom di kanan atas agar rapi
        L.control.zoom({ position: 'topright' }).addTo(map);

        // Map BaseLayer - Dark Mode
        var baseLayer = L.tileLayer('https://{s}.basemaps.cartocdn.com/dark_all/{z}/{x}/{y}{r}.png', {
            attribution: '&copy; OpenStreetMap &copy; CARTO'
        }).addTo(map);

        //==========================================================
        // 2. FORMALISASI SIMULASI BOUNDARY KABUPATEN BANGKA BARAT
        //==========================================================
        // Geometri kasar poligon wilayah Bangka Barat untuk visualisasi kliping wilayah
        var bangkaBaratGeoJSON = {
            "type": "Feature",
            "properties": { "name": "Bangka Barat" },
            "geometry": {
                "type": "Polygon",
                "coordinates": [[
                    [105.15, -1.60], [105.40, -1.55], [105.75, -1.68], 
                    [105.90, -1.95], [105.65, -2.10], [105.35, -2.05], 
                    [105.08, -1.85], [105.15, -1.60]
                ]]
            }
        };

        var boundaryLayer = L.geoJSON(bangkaBaratGeoJSON, {
            style: {
                color: '#ef4444', // Red border dari codingan GEE
                weight: 2,
                fillColor: 'transparent',
                dashArray: '4'
            }
        }).addTo(map);

        // Toggle Boundary
        document.getElementById('layer-boundary').addEventListener('change', function(e) {
            if(e.target.checked) map.addLayer(boundaryLayer);
            else map.removeLayer(boundaryLayer);
        });

        //==========================================================
        // 3. LOGIKA RASTER CITRA OVERLAY (SIMULASI RASTER GEE)
        //==========================================================
        // Batas cakupan gambar raster (Bounding Box Bangka Barat)
        var imageBounds = [[-2.15, 105.00], [-1.50, 106.00]];

        // Inisialisasi Layer Gambar dengan warna placeholder sesuai fungsionalitas menu GEE
        var rasterLayers = {
            comp24: L.rectangle(imageBounds, {color: "#1e3a8a", weight: 0, fillOpacity: 0.6}), // Simulasi Citra 2024
            comp25: L.rectangle(imageBounds, {color: "#0f766e", weight: 0, fillOpacity: 0.6}), // Simulasi Citra 2025
            ndvi24: L.rectangle(imageBounds, {color: "#166534", weight: 0, fillOpacity: 0.7}),
            ndvi25: L.rectangle(imageBounds, {color: "#14532d", weight: 0, fillOpacity: 0.7}),
            ndmi24: L.rectangle(imageBounds, {color: "#0369a1", weight: 0, fillOpacity: 0.6}),
            ndmi25: L.rectangle(imageBounds, {color: "#075985", weight: 0, fillOpacity: 0.6}),
            raster24: L.rectangle(imageBounds, {color: "#b91c1c", weight: 0, fillOpacity: 0.7}),
            rf24: L.rectangle(imageBounds, {color: "#15803d", weight: 0, fillOpacity: 0.7}),
            rf25: L.rectangle(imageBounds, {color: "#166534", weight: 0, fillOpacity: 0.7}),
            changedetect: L.rectangle(imageBounds, {color: "#2563eb", weight: 0, fillOpacity: 0.75})
        };

        // Fungsi membersihkan layer aktif sebelum memuat layer baru
        function clearRasterGroup(groupName) {
            if (groupName === 'composite') {
                map.removeLayer(rasterLayers.comp24);
                map.removeLayer(rasterLayers.comp25);
            } else if (groupName === 'class') {
                map.removeLayer(rasterLayers.raster24);
                map.removeLayer(rasterLayers.rf24);
                map.removeLayer(rasterLayers.rf25);
                hideLegend();
            }
        }

        // Listener Radio Kancing untuk Composite 
        document.getElementById('layer-comp24').addEventListener('change', function() { clearRasterGroup('composite'); map.addLayer(rasterLayers.comp24); });
        document.getElementById('layer-comp25').addEventListener('change', function() { clearRasterGroup('composite'); map.addLayer(rasterLayers.comp25); });
        document.getElementById('layer-comp-off').addEventListener('change', function() { clearRasterGroup('composite'); });

        // Listener Tombol Feature Stack (NDVI & NDMI)
        var featureButtons = ['ndvi24', 'ndvi25', 'ndmi24', 'ndmi25'];
        featureButtons.forEach(function(id) {
            document.getElementById('btn-' + id).addEventListener('click', function() {
                // Hapus semua indeks spektra dulu
                featureButtons.forEach(function(b) { map.removeLayer(rasterLayers[b]); });
                map.addLayer(rasterLayers[id]);
                showIndexLegend(id.toUpperCase());
            });
        });

        // Listener Radio Kancing untuk Hasil Klasifikasi
        document.getElementById('layer-raster24').addEventListener('change', function() { clearRasterGroup('class'); map.addLayer(rasterLayers.raster24); showClassLegend("Klasifikasi NDVI 2024"); });
        document.getElementById('layer-rf24').addEventListener('change', function() { clearRasterGroup('class'); map.addLayer(rasterLayers.rf24); showClassLegend("Random Forest 2024"); });
        document.getElementById('layer-rf25').addEventListener('change', function() { clearRasterGroup('class'); map.addLayer(rasterLayers.rf25); showClassLegend("Random Forest 2025"); });
        document.getElementById('layer-class-off').addEventListener('change', function() { clearRasterGroup('class'); });

        // Change Detection Map Toggle
        document.getElementById('layer-changedetect').addEventListener('change', function(e) {
            if(e.target.checked) {
                map.addLayer(rasterLayers.changedetect);
                showChangeLegend();
            } else {
                map.removeLayer(rasterLayers.changedetect);
                hideLegend();
            }
        });

        // Kontrol Opacity Slider global untuk semua layer simulasi raster
        document.getElementById('opacity-slider').addEventListener('input', function(e) {
            var val = e.target.value;
            Object.keys(rasterLayers).forEach(function(key) {
                rasterLayers[key].setStyle({fillOpacity: val});
            });
        });

        //==========================================================
        // 4. GENERATOR TITIK DATA TRAINING (STRATIFIED SAMPLE)
        //==========================================================
        var sampleLayerGroup = L.layerGroup();
        
        // Buat simulasi sebaran acak 75 titik di dalam boundary Bangka Barat
        function generateSamplePoints() {
            for(var i=0; i<60; i++) {
                var lat = -1.65 - (Math.random() * 0.35);
                var lng = 105.15 + (Math.random() * 0.65);
                var isVegetasi = Math.random() > 0.4;
                
                var markerColor = isVegetasi ? '#22c55e' : '#ef4444'; // Hijau (Veg), Merah (Non-Veg)
                var className = isVegetasi ? 'Kelas: Vegetasi (1)' : 'Kelas: Non-Vegetasi (0)';

                var marker = L.circleMarker([lat, lng], {
                    radius: 5,
                    fillColor: markerColor,
                    color: '#ffffff',
                    weight: 1,
                    fillOpacity: 1
                }).bindPopup(`<b>Training Point Data</b><br>${className}<br>Lat: ${lat.toFixed(4)}, Lng: ${lng.toFixed(4)}`);
                
                sampleLayerGroup.addLayer(marker);
            }
        }
        generateSamplePoints();

        document.getElementById('layer-samples').addEventListener('change', function(e) {
            if(e.target.checked) map.addLayer(sampleLayerGroup);
            else map.removeLayer(sampleLayerGroup);
        });

        //==========================================================
        // 5. SISTEM LEGENDA DINAMIS
        //==========================================================
        var legendEl = document.getElementById('map-legend');
        var legendTitle = document.getElementById('legend-title');
        var legendItems = document.getElementById('legend-items');

        function hideLegend() { legendEl.classList.add('hidden'); }

        function showIndexLegend(name) {
            legendTitle.innerText = name;
            legendItems.innerHTML = `
                <div class="flex items-center space-x-2"><span class="w-7 h-3 inline-block" style="background:linear-gradient(to right, white, yellow, green, darkgreen)"></span><span class="text-[10px]">-0.2 s/d 0.8</span></div>
            `;
            legendEl.classList.remove('hidden');
        }

        function showClassLegend(title) {
            legendTitle.innerText = title;
            legendItems.innerHTML = `
                <div class="flex items-center space-x-2"><span class="w-3 h-3 bg-emerald-600 inline-block rounded-sm"></span><span>Vegetasi</span></div>
                <div class="flex items-center space-x-2"><span class="w-3 h-3 bg-red-650 inline-block rounded-sm"></span><span>Non-Vegetasi</span></div>
            `;
            legendEl.classList.remove('hidden');
        }

        function showChangeLegend() {
            legendTitle.innerText = "Deteksi Perubahan";
            legendItems.innerHTML = `
                <div class="flex items-center space-x-2"><span class="w-3 h-3 bg-red-600 inline-block rounded-sm"></span><span>Tetap Non-Vegetasi</span></div>
                <div class="flex items-center space-x-2"><span class="w-3 h-3 bg-blue-600 inline-block rounded-sm"></span><span>Gain Vegetasi (Tumbuh)</span></div>
                <div class="flex items-center space-x-2"><span class="w-3 h-3 bg-yellow-450 inline-block rounded-sm"></span><span>Loss Vegetasi (Hilang)</span></div>
                <div class="flex items-center space-x-2"><span class="w-3 h-3 bg-green-600 inline-block rounded-sm"></span><span>Tetap Vegetasi</span></div>
            `;
            legendEl.classList.remove('hidden');
        }
    </script>
